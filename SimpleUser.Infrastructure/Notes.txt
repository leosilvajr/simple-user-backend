# 1. Criar a solução
dotnet new sln -n SimpleUser -o C:\Users\Leonardo\Desktop\simple-user

# 2. Criar os projetos
dotnet new webapi -n SimpleUser.API -o C:\Users\Leonardo\Desktop\simple-user\SimpleUser.API
dotnet new classlib -n SimpleUser.Application -o C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Application
dotnet new classlib -n SimpleUser.Domain -o C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Domain
dotnet new classlib -n SimpleUser.Infrastructure -o C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Infrastructure

# 3. Adicionar os projetos à solução
dotnet sln C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln add C:\Users\Leonardo\Desktop\simple-user\SimpleUser.API\SimpleUser.API.csproj
dotnet sln C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln add C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Application\SimpleUser.Application.csproj
dotnet sln C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln add C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Domain\SimpleUser.Domain.csproj
dotnet sln C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln add C:\Users\Leonardo\Desktop\simple-user\SimpleUser.Infrastructure\SimpleUser.Infrastructure.csproj

# 4. Restaurar dependências
dotnet restore C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln

# 5. Abrir a solução no Visual Studio
start C:\Users\Leonardo\Desktop\simple-user\SimpleUser.sln

#6. Instalar EntityFrameworkCore
dotnet add SimpleUser.Infrastructure package Microsoft.EntityFrameworkCore.SqlServer
dotnet add SimpleUser.Infrastructure package Microsoft.EntityFrameworkCore.Design
dotnet add SimpleUser.Infrastructure package Microsoft.EntityFrameworkCore.Tools
dotnet add SimpleUser.API package Microsoft.EntityFrameworkCore.Design

#7. Criar uma Migration (opcional)
dotnet ef migrations add InitialCreate -p SimpleUser.Infrastructure -s SimpleUser.API
dotnet ef database update -p SimpleUser.Infrastructure -s SimpleUser.API


#8. Adicionar Swashbuckle.AspNetCore
dotnet add SimpleUser.API package Swashbuckle.AspNetCore
